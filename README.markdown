The scale7 maven repo.

To use add the following to your pom.xml:

    <repository>
        <id>maven.scale7.org</id>
        <name>Scale7 Maven Repo</name>
        <url>https://github.com/s7/mvnrepo/raw/master</url>
        <snapshots>
            <enabled>true</enabled>
        </snapshots>
    </repository>
